EmissionsProcess.sql

Resource: `CaisToNodeCore.Models.Cais.CaisSQL.EmissionsProcess

-- EmissionsProcess.sql
-- DECLARE @InventoryYear CHAR(4) = Year(Getdate()) - 1;-- Uncomment for testing

WITH DataYears
AS (
    SELECT DISTINCT PY.pla_id
        , Higest.plyear_id
        , Higest.plyear
    FROM plyear PY
    CROSS APPLY (
        SELECT TOP 1 plyear_id
            , plyear
        FROM plyear
        WHERE pla_id = PY.pla_id
            AND plyear <= @InventoryYear
        ORDER BY plyear DESC
        ) AS Higest
    )
SELECT 'CO00000008' + PL.PLA_A_CNTY + '0'
            + RIGHT('000' + Ltrim(Rtrim(PL.PLA_A_PLNT)), 4) AS FacilitySiteIdentifier
       ,PL.PLA_A_CNTY + '-' + RIGHT('000' + Ltrim(Rtrim(PL.PLA_A_PLNT)), 4)
            + '-' + RIGHT('000' + Ltrim(Rtrim(PT.POI_A_PNT)), 3) AS UnitIdentifier
       ,PL.PLA_A_CNTY + '-' + RIGHT('000' + LTRIM(RTRIM(PL.PLA_A_PLNT)),4) 
            + '-' + RIGHT('000' + LTRIM(RTRIM(PT.POI_A_PNT)),3) + '-' 
            + RIGHT('00' + LTRIM(RTRIM(PR.pro_afsid)), 2) AS ProcessIdentifier
       ,PL.PLA_A_CNTY + '-' + RIGHT('000' + LTRIM(RTRIM(PL.PLA_A_PLNT)),4) 
            + '-S' + RIGHT('000' + LTRIM(RTRIM(ST.stk_afsid)),3) AS ReleasePointIdentifier
       ,PR.pro_scc AS SourceClassificationCode
       ,RTRIM(PR.pro_desc) AS ProcessDescription
FROM cais..Plant AS PL
       JOIN  cais..Point AS PT 
         ON PT.POI_PLA_ID = PL.PLA_ID
       JOIN DataYears AS DY
         ON DY.pla_id = PL.PLA_ID
       JOIN cais..PointHis AS PH
         ON PH.poi_id = PT.POI_ID 
            AND PH.plyear_id = DY.plyear_id
      JOIN CAIS..process AS PR
         ON PR.poh_id = PH.poh_id
      LEFT JOIN dbo.Stacks AS ST                                  
         ON PR.stkyear_id = ST.stkyear_id            
WHERE PL.PLA_A_CNTY NOT IN ('000', '778', '888')
ORDER BY UnitIdentifier
