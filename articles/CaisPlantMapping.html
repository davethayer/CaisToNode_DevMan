FacilitySite.sql

The first line may be uncommented to facilitate interactive testing of the query:

Resource: CaisToNodeCore.Models.Cais.CaisSQL.FacilitySite

-- DECLARE @InventoryYear CHAR(4) = Year(Getdate()) - 1; -- Uncomment for testing
SELECT 'CO00000008' + PL.PLA_A_CNTY + '0'
            + RIGHT('000' + Trim(PL.PLA_A_PLNT), 4) AS FacilitySiteIdentifier
       , PL.PLA_A_CNTY AS CountyFipsCode
       , Trim(PL.PLA_NAME) AS FacilitySiteName
       , PL.PLA_OPSTAT AS CaisPlantOperatingStatus
       , Trim(PL.PLA_NAICSP) AS NAICSCode
       , Trim(AD.Addr1) AS LocationAddressText
       , Trim(AD.Addr2) AS SupplementalLocationText
       , Trim(COALESCE(CI.CIT_NAME, AD.City)) AS LocalityName
       , Trim(AD.State) AS LocationAddressStateCode
       , Trim(AD.Zip) AS LocationAddressPostalCode
       , Coalesce(AD.CountyCd, PL.PLA_A_CNTY) AS LocationAddressCountyFipsCode
       , -(Floor(PL.pla_long / 10000) + (Floor(PL.pla_long / 100) % 100) / 60.0 + (PL.pla_long % 100) / 3600.0) AS LongitudeMeasure
       , Floor(PL.pla_lat / 10000) + (Floor(PL.pla_lat / 100) % 100) / 60.0 + (PL.pla_lat % 100) / 3600.0 AS LatitudeMeasure
   FROM CAIS..vPlantCMS AS PL
    INNER JOIN CAIS..AddrPla AS ADP
     ON  PL.Pla_ID = ADP.Pla_ID
    INNER JOIN CAIS..Addresses AS AD
     ON  AD.Address_ID = ADP.Address_ID
    LEFT JOIN CAIS..CITY AS CI
     ON  CI.CIT_FIPS = AD.CityCd
    LEFT JOIN CAIS..COUNTY AS CO
     ON  CO.COU_CODE = PL.PLA_A_CNTY
    LEFT JOIN CAIS..plyear AS PY
         ON  pl.pla_id = PY.pla_id
        AND PY.plyear = (SELECT MAX(PY1.plyear)
                        FROM   CAIS..plyear AS PY1
                        WHERE  PY1.pla_id = PL.pla_id
                            AND PY1.plyear <= @InventoryYear)
WHERE  CO.COU_FIPS = CO.COU_AIRSID -- No land development pseudo-counties
    AND PL.PLA_A_CNTY NOT IN ( '000', '888' ) -- No other pseudo-counties
    AND ADP.Type_ID = (SELECT Type_ID
    FROM CAIS..ContactType AS CT
    WHERE  TypeCode = 'PLNTLOC'
        AND ActiveFl = 1)
ORDER BY FacilitySiteIdentifier ASC
